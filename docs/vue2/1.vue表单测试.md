::: demo input 输入正整数(正则)

```vue
<template>
    <div>
        <input v-model="formValue" @input="onInput" placeholder="请输入正整数" />
    </div>
</template>

<script setup>
import { ref } from "vue"

const formValue = ref("")

const onInput = (e) => {
    formValue.value = e.target.value.replace(/[^\d]/g, "")
}
</script>
```

const formValue = ref("")
